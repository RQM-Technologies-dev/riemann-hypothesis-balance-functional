\begin{lemma}[Nonvanishing factors in the strip]
\label{lem:strip-factors-nonzero}
If $0<\RePart(s)<1$, then $\Gamma(s)\neq 0$ and $1-s\neq 0$.
\end{lemma}
\noindent\emph{Status: standard.}

\begin{lemma}[Strip zero correspondence]
\label{lem:strip-zero-correspondence}
If $0<\RePart(s)<1$, then
\[
\Phi(s)=0\iff\zeta(s)=0.
\]
\end{lemma}
\begin{proof}
By \cref{lem:strip-factors-nonzero}, the factor $\Gamma(s)(1-s)$ is nonzero in the strip.
So $\Gamma(s)(1-s)\zeta(s)=0$ is equivalent to $\zeta(s)=0$.
\end{proof}
