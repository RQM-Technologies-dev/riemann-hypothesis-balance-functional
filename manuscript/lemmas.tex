\subsection*{A. Standard symmetry inputs}

\begin{lemma}[Reflection symmetry of nontrivial zeros]
If $\rho$ is a nontrivial zero of $\zeta$, then $1-\rho$ is also a nontrivial zero.
\end{lemma}
\noindent\emph{Status: standard.} Follows from the functional equation for completed zeta.

\begin{lemma}[Conjugation symmetry of nontrivial zeros]
If $\rho$ is a nontrivial zero of $\zeta$, then $\overline{\rho}$ is also a nontrivial zero.
\end{lemma}
\noindent\emph{Status: standard.} Follows from Schwarz reflection.
