This section states the shortest main theorem targets built on one object:
\[
\Phi(q):=\Gamma(q)(1-q)\zeta(q).
\]

\subsection*{Established input (standard)}
\begin{proposition}[Functional symmetry of completed zeta]
\label{prop:functional-symmetry}
The completed zeta function satisfies $\xi(s)=\xi(1-s)$; see \cite{Riemann1859,Edwards1974,Titchmarsh1986}.
\end{proposition}

\subsection*{Step 1: Canonical balance set}
In this manuscript, balance means exactly
\[
\Phi(q)=0.
\]
Define
\[
\mathcal{B}:=\{q\in\C:0<\RePart(q)<1,\ \Phi(q)=0\}.
\]

\subsection*{Step 4: Conditional RH reduction}
\begin{proposition}[Conditional RH reduction from Targets A and B]
\label{prop:rh-reduction-main-spine}
If Targets~A and~B hold, then every nontrivial zero of $\zeta$ has real part $\tfrac12$.
\end{proposition}

\begin{proof}
By Target~A, each nontrivial zero lies in $\mathcal{B}$.
By Target~B, each point of $\mathcal{B}$ satisfies $\RePart(q)=\tfrac12$.
Therefore every nontrivial zero lies on the critical line.
\end{proof}

\begin{remark}[Status discipline]
The RH deduction above is conditional.
This manuscript does not claim that Target~A or Target~B is already proved.
\end{remark}
