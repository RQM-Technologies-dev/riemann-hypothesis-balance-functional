\subsection*{Canonical setup}
Define
\[
\Phi(s):=\Gamma(s)(1-s)\zeta(s).
\]

\subsection*{Result 1: strip correspondence}
\begin{proposition}[Zero correspondence in the strip]
\label{prop:strip-correspondence}
For every $s\in\C$ with $0<\RePart(s)<1$,
\[
\Phi(s)=0\iff \zeta(s)=0.
\]
\end{proposition}

\begin{proof}
If $0<\RePart(s)<1$, then $s$ is not a nonpositive integer, so $\Gamma(s)\neq 0$.
Also $1-s\neq 0$ because $\RePart(s)\neq 1$.
Hence $\Gamma(s)(1-s)$ is nonzero, and multiplying by this factor preserves the zero set.
Therefore $\Phi(s)=0\iff\zeta(s)=0$.
\end{proof}

\subsection*{Result 3: conditional RH reduction}
\begin{proposition}[Conditional RH reduction]
\label{prop:rh-reduction-main-spine}
If \cref{thm:core-target} holds, then RH follows.
\end{proposition}

\begin{proof}
Let $s$ be a nontrivial zero of $\zeta$.
Then $0<\RePart(s)<1$ by definition of nontrivial zero, and \cref{prop:strip-correspondence} gives $\Phi(s)=0$.
Applying \cref{thm:core-target}, we obtain $\RePart(s)=\tfrac12$.
Thus every nontrivial zero lies on the critical line.
\end{proof}
