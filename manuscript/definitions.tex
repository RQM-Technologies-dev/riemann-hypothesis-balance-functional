This section fixes the canonical definitions used in the simplified proof attempt.

\begin{definition}[Critical strip and critical line]
The critical strip is
\[
\{q\in\C:0<\RePart(q)<1\},
\]
and the critical line is
\[
\{q\in\C:\RePart(q)=\tfrac12\}.
\]
\end{definition}

\begin{definition}[Canonical object]
\label{def:phi-canonical}
Define
\[
\Phi(q):=\Gamma(q)(1-q)\zeta(q).
\]
This is the only main-text balance object in this manuscript.
\end{definition}

\begin{definition}[Balanced point]
\label{def:balanced-point}
A point $q$ is \emph{balanced} if and only if $\Phi(q)=0$.
On the strip define
\[
\mathcal{B}:=\{q\in\C:0<\RePart(q)<1,\ \Phi(q)=0\}.
\]
\end{definition}

\begin{remark}[Immediate strip identity]
For $0<\RePart(q)<1$, $\Gamma(q)\neq0$ and $1-q\neq0$.
Hence
\[
\Phi(q)=0\iff \zeta(q)=0
\]
on the strip.
\end{remark}

\begin{remark}[Status boundary]
These are definitions and standard analytic facts.
No RH claim follows from this section alone.
\end{remark}
