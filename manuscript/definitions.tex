This section fixes the canonical definitions for the manuscript under review.

\begin{definition}[Critical strip and critical line]
The critical strip is
\[
\{s\in\C:0<\RePart(s)<1\},
\]
and the critical line is
\[
\{s\in\C:\RePart(s)=\tfrac12\}.
\]
\end{definition}

\begin{definition}[Canonical object]
\label{def:phi-canonical}
Define
\[
\Phi(s):=\Gamma(s)(1-s)\zeta(s).
\]
\end{definition}

\begin{definition}[Balanced point]
\label{def:balanced-point}
A point $s\in\C$ is \emph{balanced} if and only if
\[
\Phi(s)=0.
\]
\end{definition}

\begin{remark}[Strip correspondence]
\label{rem:strip-correspondence}
If $0<\RePart(s)<1$, then $\Gamma(s)\neq 0$ and $1-s\neq 0$. Therefore
\[
\Phi(s)=0\iff \zeta(s)=0
\]
on the strip.
\end{remark}

\begin{remark}[Status boundary]
The definitions and strip correspondence above are standard analytic facts.
They do not by themselves prove RH.
\end{remark}
